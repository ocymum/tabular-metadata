101    @Test
102    public void testCorrectParsingOfValues() {
103        for (final DataType testType : DataType.values()) {
104            for (final String testValue : sampleValues.get(testType)) {
105                final Set<DataType> result = parseableAs(testValue);
106                assertEquals("Did not find the appropriate datatypes suggested as parseable for a " + testType + "!",
107                        expectedParseableTypes
108                                .get(testType), result);
109            }
110        }
111    }
112
113    @Test
114    public void testSupertypes() {
115        for (final DataType testDatatype : DataType.values()) {
116            final Set<DataType> result = testDatatype.supertypes();
117            assertEquals("Did not find the appropriate supertypes for" + testDatatype + "!", expectedSuperTypes
118                    .get(testDatatype), result);
119        }
120    }

122    @Test
123    public void testOrderingByHierarchy() {
124        List<DataType> listToBeSorted = _(Decimal, PositiveInteger, NonNegativeInteger, Integer, String);
125        List<DataType> listInCorrectSorting =
126                _(String, Decimal, Integer, NonNegativeInteger, PositiveInteger);
127        sort(listToBeSorted, orderingByHierarchy);
128        assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, listToBeSorted);
129
130        listToBeSorted = _(Geographic, String);
131        listInCorrectSorting = _(String, Geographic);
132        sort(listToBeSorted, orderingByHierarchy);
133        assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, listToBeSorted);
134
135        listToBeSorted = _(DateTime, String);
136        listInCorrectSorting = _(String, DateTime);
137        sort(listToBeSorted, orderingByHierarchy);
138        assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, listToBeSorted);
139
140        listToBeSorted = _(Boolean, String);
141        listInCorrectSorting = _(String, Boolean);
142        sort(listToBeSorted, orderingByHierarchy);
143        assertEquals("Got wrong ordering by hierarchy of datatypes!", listInCorrectSorting, listToBeSorted);
144    }

146    @Test
147    public void testBadGeographies() {
148        String testValue = "23, 23, 23, 23";
149        assertFalse("Accepted a four-valued tuple as geographic coordinates!", parseableAs(testValue)
150                .contains(Geographic));
151        testValue = "23";
152        assertFalse("Accepted a one-valued tuple as geographic coordinates!", parseableAs(testValue)
153                .contains(Geographic));
154    }
